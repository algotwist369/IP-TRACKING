üåê Basic HTML/JavaScript Integration

Step 1: Add the tracking script to your HTML

Place this code in the <head> or <body> section of your HTML:

<script> window.IPTrackerConfig = { trackingCode: 'YOUR_TRACKING_CODE', website: 'YOUR_DOMAIN' }; </script> <script src="https://yourdomain.com/tracking-script.js"></script>

Step 2: Complete HTML Example

Here's a complete HTML page with the tracking script:

<!DOCTYPE html> <html lang="en"> <head> <meta charset="UTF-8"> <meta name="viewport" content="width=device-width, initial-scale=1.0"> <title>Your Website</title> <!-- IP Tracker Configuration --> <script> window.IPTrackerConfig = { trackingCode: 'YOUR_TRACKING_CODE', website: 'YOUR_DOMAIN' }; </script> <script src="https://yourdomain.com/tracking-script.js"></script> </head> <body> <h1>Welcome to Your Website</h1> <p>This page is being tracked by IP Tracker</p> <!-- Test tracking manually --> <button onclick="testTracking()">Test Tracking</button> <script> function testTracking() { if (window.IPTracker) { window.IPTracker.trackEvent('button_click', { button: 'test_button', page: 'homepage' }); alert('Tracking event sent!'); } } </script> </body> </html>
‚úÖ That's it! Your website is now being tracked. The script will automatically:
  • Collect visitor information when the page loads
  • Track user interactions and behavior
  • Detect VPNs, proxies, and suspicious activity
  • Send data to your IP Tracker dashboard

‚öõÔ∏è React Integration

Step 1: Install the tracking script in your React app

Create a custom hook for IP tracking:

// hooks/useIPTracker.js import { useEffect } from 'react'; export const useIPTracker = (trackingCode, website) => { useEffect(() => { // Load IP Tracker script const script = document.createElement('script'); script.innerHTML = \` window.IPTrackerConfig = { trackingCode: '${trackingCode}', website: '${website}' }; \`; document.head.appendChild(script); const trackingScript = document.createElement('script'); trackingScript.src = 'https://yourdomain.com/tracking-script.js'; trackingScript.onload = () => { console.log('IP Tracker loaded successfully'); }; document.head.appendChild(trackingScript); // Cleanup on unmount return () => { document.head.removeChild(script); document.head.removeChild(trackingScript); }; }, [trackingCode, website]); };

Step 2: Use the hook in your components

// components/App.js import React from 'react'; import { useIPTracker } from './hooks/useIPTracker'; function App() { // Initialize IP tracking useIPTracker('YOUR_TRACKING_CODE', 'YOUR_DOMAIN'); const handleButtonClick = () => { if (window.IPTracker) { window.IPTracker.trackEvent('button_click', { component: 'App', action: 'button_click' }); } }; return ( <div className="App"> <h1>My React App</h1> <button onClick={handleButtonClick}> Click Me (Tracked) </button> </div> ); } export default App;

Step 3: Track specific events in components

// components/ProductCard.js import React from 'react'; function ProductCard({ product }) { const handleView = () => { if (window.IPTracker) { window.IPTracker.trackEvent('product_view', { productId: product.id, productName: product.name, category: product.category }); } }; const handlePurchase = () => { if (window.IPTracker) { window.IPTracker.trackEvent('purchase', { productId: product.id, amount: product.price, currency: 'USD' }); } }; useEffect(() => { handleView(); // Track when product is viewed }, [product.id]); return ( <div className="product-card"> <h3>{product.name}</h3> <p>${product.price}</p> <button onClick={handlePurchase}> Buy Now </button> </div> ); }

üññ Vue.js Integration

Step 1: Create a Vue plugin for IP tracking

// plugins/ipTracker.js export default { install(app, options) { const { trackingCode, website } = options; // Load IP Tracker script const script = document.createElement('script'); script.innerHTML = \` window.IPTrackerConfig = { trackingCode: '${trackingCode}', website: '${website}' }; \`; document.head.appendChild(script); const trackingScript = document.createElement('script'); trackingScript.src = 'https://yourdomain.com/tracking-script.js'; document.head.appendChild(trackingScript); // Add global methods app.config.globalProperties.$trackEvent = (eventName, data) => { if (window.IPTracker) { window.IPTracker.trackEvent(eventName, data); } }; // Add to provide/inject app.provide('trackEvent', app.config.globalProperties.$trackEvent); } };

Step 2: Use the plugin in your main.js

// main.js import { createApp } from 'vue'; import App from './App.vue'; import ipTracker from './plugins/ipTracker'; const app = createApp(App); app.use(ipTracker, { trackingCode: 'YOUR_TRACKING_CODE', website: 'YOUR_DOMAIN' }); app.mount('#app');

Step 3: Use tracking in your Vue components

<template> <div class="product-page"> <h1>{{ product.name }}</h1> <button @click="handlePurchase">Buy Now</button> </div> </template> <script> export default { name: 'ProductPage', data() { return { product: { id: 1, name: 'Sample Product', price: 99.99 } }; }, mounted() { // Track page view this.$trackEvent('page_view', { page: 'product', productId: this.product.id }); }, methods: { handlePurchase() { this.$trackEvent('purchase', { productId: this.product.id, amount: this.product.price }); } } }; </script>

Step 1: Add tracking code to your theme

Add this code to your theme's functions.php file:

<?php // Add IP Tracker to WordPress function add_ip_tracker() { ?> <script> window.IPTrackerConfig = { trackingCode: 'YOUR_TRACKING_CODE', website: 'YOUR_DOMAIN' }; </script> <script src="https://yourdomain.com/tracking-script.js"></script> <?php } add_action('wp_head', 'add_ip_tracker'); ?>

Step 2: Alternative: Add via WordPress Customizer

For easier management, you can add it through the WordPress Customizer:

<?php // Add custom field to WordPress Customizer function ip_tracker_customizer($wp_customize) { $wp_customize->add_section('ip_tracker', array( 'title' => 'IP Tracker Settings', 'priority' => 30, )); $wp_customize->add_setting('tracking_code', array( 'default' => '', 'sanitize_callback' => 'sanitize_text_field', )); $wp_customize->add_control('tracking_code', array( 'label' => 'Tracking Code', 'section' => 'ip_tracker', 'type' => 'text', )); } add_action('customize_register', 'ip_tracker_customizer'); // Output tracking code function output_ip_tracker() { $tracking_code = get_theme_mod('tracking_code'); if (!empty($tracking_code)) { ?> <script> window.IPTrackerConfig = { trackingCode: '<?php echo esc_js($tracking_code); ?>', website: '<?php echo esc_js(get_site_url()); ?>' }; </script> <script src="https://yourdomain.com/tracking-script.js"></script> <?php } } add_action('wp_head', 'output_ip_tracker'); ?>

Privacy-First Tracking

Implement privacy-conscious tracking:

// Privacy-first tracking implementation function initializePrivacyTracking() { // Check for cookie consent if (!hasCookieConsent()) { return; } // Check for DNT (Do Not Track) if (navigator.doNotTrack === '1') { console.log('DNT enabled, tracking disabled'); return; } // Initialize with privacy settings window.IPTrackerConfig = { trackingCode: 'YOUR_TRACKING_CODE', website: 'YOUR_DOMAIN', privacyMode: 'enhanced', cookieConsent: true, gdprCompliant: true, enableDeviceFingerprinting: false, // Disable for privacy enableBehavioralTracking: false // Disable for privacy }; // Load tracking script loadTrackingScript(); } function hasCookieConsent() { // Implement your cookie consent check return localStorage.getItem('cookieConsent') === 'accepted'; } function loadTrackingScript() { const script = document.createElement('script'); script.src = 'https://yourdomain.com/tracking-script.js'; document.head.appendChild(script); }

Performance Optimization

Optimize tracking for better performance:

// Lazy load tracking for better performance function lazyLoadTracking() { // Only load tracking when user interacts let trackingLoaded = false; function loadTracking() { if (trackingLoaded) return; window.IPTrackerConfig = { trackingCode: 'YOUR_TRACKING_CODE', website: 'YOUR_DOMAIN' }; const script = document.createElement('script'); script.src = 'https://yourdomain.com/tracking-script.js'; document.head.appendChild(script); trackingLoaded = true; } // Load on first user interaction document.addEventListener('click', loadTracking, { once: true }); document.addEventListener('scroll', loadTracking, { once: true }); // Or load after a delay setTimeout(loadTracking, 3000); } // Initialize lazy loading lazyLoadTracking();
